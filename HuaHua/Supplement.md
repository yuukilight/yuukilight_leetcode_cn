# unordered_map
简单的使用
```c++
#include <unordered_map>
using namespace std;
// 创建
unordered_map<int, int> indices;
// 基本函数
indices.count(end);
indices.begin();
indices.end();
indices.find(key);
indices.empty();
indices.clear();

erase;
emplace;

// map 创建与赋值
// 如果访问元素不存在会返回 0
map<char, int> dx{{'L',-1},{'R',1}},dy{{'U',1},{'D',-1}};
```
